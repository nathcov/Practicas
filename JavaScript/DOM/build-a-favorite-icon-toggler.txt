** start of script.js **

const botones = document.querySelectorAll(".favorite-icon");

botones.forEach((boton) => {
  boton.addEventListener("click", () => {
boton.classList.toggle("filled");

  boton.innerHTML = boton.classList.contains("filled") ? "&#10084;" : "&#9825;";
  });
})
