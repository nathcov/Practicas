** start of script.js **

const footballTeam = {
  team: "Real Madrid",
  year: 2024,
  headCoach: "Carlo Ancelotti",
  players: [
    { name: "Thibaut Courtois", position: "goalkeeper", isCaptain: false },
    { name: "Dani Carvajal", position: "defender", isCaptain: true },
    { name: "Luka Modrić", position: "midfielder", isCaptain: false },
    { name: "Vinícius Júnior", position: "forward", isCaptain: false }
  ]
};

const selectContainer = document.querySelector("#players");
const productsContainer = document.querySelector("#player-cards");

// Mostrar información del equipo
document.getElementById("head-coach").textContent = footballTeam.headCoach;
document.getElementById("team").textContent = footballTeam.team;
document.getElementById("year").textContent = footballTeam.year;

function playerCards(playerPosition = "all") {
  const filteredPlayers =
    playerPosition === "all"
      ? footballTeam.players
      : footballTeam.players.filter(
          ({ position }) => position === playerPosition
        );

  return filteredPlayers
    .map(({ name, position, isCaptain }) => {
      return `
          <div class="player-card">
            <h2>${isCaptain ? "(Captain) " : ""}${name}</h2>
            <p>Position: ${position}</p>
          </div>
        `;
    })
    .join("");
}

// Carga inicial de todos los jugadores
productsContainer.innerHTML = playerCards("all");

selectContainer.addEventListener("change", (e) => {
  productsContainer.innerHTML = playerCards(e.target.value);
});

** end of script.js **
