function sumPrimes (num) {
  if (num < 2) {
    return 0;
  } else {
    let numeros = [];
    for (let i = 2; i <= num; i++) {
      numeros.push(i);
    }

    return numeros.filter(n => {
      for (let i = 2; i < n-1; i++) {
        if (n % i === 0) {
          return false;
        } 
      }
      return true;
    }).reduce((acumulador, valorActual) => acumulador + valorActual, 0);
  }
}
