function chunkArrayInGroups (arr, num) {
  let resultado = [];
  for (let i = 0; i < arr.length; i += num) {
    let corte = arr.slice(i, i + num);
    resultado.push(corte);
  }
  return resultado;
}

console.log(chunkArrayInGroups(["a", "b", "c", "d"], 2));
