function repeatStringNumTimes(string, num) {
  let arreglo = [];
  if (num <= 0) {
    return "";
  } else {
    for (let i = 0; i < num; i++) {
      arreglo.push(string);
    }
  }
  return arreglo.join("");
}

console.log(repeatStringNumTimes("*", 3));
