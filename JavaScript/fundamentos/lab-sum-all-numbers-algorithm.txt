function sumAll(array) {
  let menor, mayor;
  if (array[0] < array[1]) {
    menor = array[0];
    mayor = array[1];
  } else {
    menor = array[1];
    mayor = array[0];
  }
  let suma = 0;
  for (let i = menor; i <= mayor; i++) {
    suma += i;
  }
  return suma;
}

console.log(sumAll([1, 4]))
