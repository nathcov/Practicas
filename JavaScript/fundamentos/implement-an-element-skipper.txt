function dropElements(arr, func) {
  for (let i = 0; i < arr.length; i++) {
    if (func(arr[i])) {
      // En cuanto encontramos el primero que cumple, 
      // cortamos desde ahí hasta el final.
      return arr.slice(i);
    }
  }
  // Si terminamos el ciclo y ninguno cumplió, devolvemos vacío.
  return [];
}
