function steamrollArray(arr) {
  // 1. Creamos un arreglo vacío donde iremos guardando los elementos "liberados"
  let result = [];

  // 2. Recorremos el arreglo original elemento por elemento
  for (let i = 0; i < arr.length; i++) {
    
    if (Array.isArray(arr[i])) {
      // 3. CASO RECURSIVO: Si el elemento es un arreglo, 
      // llamamos a la función de nuevo con ese sub-arreglo.
      // Usamos el operador spread (...) para meter los elementos aplanados en 'result'
      result.push(...steamrollArray(arr[i]));
    } else {
      // 4. CASO BASE: Si NO es un arreglo (es un número, objeto, string, etc.)
      // simplemente lo metemos al resultado.
      result.push(arr[i]);
    }
  }

  // 5. Devolvemos el arreglo ya aplanado
  return result;
}
