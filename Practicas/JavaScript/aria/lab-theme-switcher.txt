** start of script.js **

const btn = document.getElementById("theme-switcher-button");
const menu = document.getElementById("theme-dropdown");

btn.addEventListener("click", () => {
    const expanded = btn.getAttribute("aria-expanded") === "true";

    btn.setAttribute("aria-expanded", String(!expanded));
    menu.hidden = expanded;
});

const themes = [
{ name:"light",
message: "¡Que se haga la luz!"
}, 
{ name: "dark",
message: "Oscuro como tu alma..."
}];

const temas = document.querySelectorAll(`[role="menuitem"]`);

const accessibilityMessage = document.querySelector('[aria-live="polite"]');

temas.forEach((temaElemento, index) => {
    temaElemento.addEventListener("click", () => {
        const selectedThemeObj = themes[index]; 
        document.body.className = `theme-${selectedThemeObj.name}`;

        accessibilityMessage.textContent = selectedThemeObj.message;
    });
});
